   357тЖТ   357тЖТ// Lightweight markdown renderer тАФ handles **bold**, `code`, and bullet lists
   358тЖТ   358тЖТfunction MarkdownLite({ content }: { content: string }) {
   359тЖТ   359тЖТ  const lines = content.split('\n')
   360тЖТ   360тЖТ  return (
   361тЖТ   361тЖТ    <div className="space-y-1">
   362тЖТ   362тЖТ      {lines.map((line, i) => {
   363тЖТ   363тЖТ        if (line.trim().startsWith('тАв ') || line.trim().startsWith('- ')) {
   364тЖТ   364тЖТ          return (
   365тЖТ   365тЖТ            <div key={i} className="flex gap-2 pl-1">
   366тЖТ   366тЖТ              <span className="text-emerald-500 font-bold">тАв</span>
   367тЖТ   367тЖТ              <span>{renderInline(line.trim().slice(2))}</span>
   368тЖТ   368тЖТ            </div>
   369тЖТ   369тЖТ          )
   370тЖТ   370тЖТ        }
   371тЖТ   371тЖТ        return <div key={i}>{renderInline(line) || <br />}</div>
   372тЖТ   372тЖТ      })}
   373тЖТ   373тЖТ    </div>
   374тЖТ   374тЖТ  )
   375тЖТ   375тЖТ}

   377тЖТ   377тЖТfunction renderInline(text: string) {
   378тЖТ   378тЖТ  if (!text) return null
   379тЖТ   379тЖТ  const parts = text.split(/(\*\*[^*]+\*\*|`[^`]+`)/g)
   380тЖТ   380тЖТ  return parts.map((p, i) => {
   381тЖТ   381тЖТ    if (p.startsWith('**') && p.endsWith('**')) {
   382тЖТ   382тЖТ      return <strong key={i} className="font-semibold">{p.slice(2, -2)}</strong>
   383тЖТ   383тЖТ    }
   384тЖТ   384тЖТ    if (p.startsWith('`') && p.endsWith('`')) {
   385тЖТ   385тЖТ      return <code key={i} className="px-1.5 py-0.5 rounded bg-background/60 text-xs font-mono">{p.slice(1, -1)}</code>
   386тЖТ   386тЖТ    }
   387тЖТ   387тЖТ    return <span key={i}>{p}</span>
   388тЖТ   388тЖТ  })
   389тЖТ   389тЖТ}

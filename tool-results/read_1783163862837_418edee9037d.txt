    67→function loadUser(): User | null {
    68→  try {
    69→    const raw = typeof window !== 'undefined' ? localStorage.getItem(USER_KEY) : null
    70→    return raw ? (JSON.parse(raw) as User) : null
    71→  } catch {
    72→    return null
    73→  }
    74→}
    75→function saveUser(u: User | null) {
    76→  if (typeof window === 'undefined') return
    77→  if (u) localStorage.setItem(USER_KEY, JSON.stringify(u))
    78→  else localStorage.removeItem(USER_KEY)
    79→}

   224→// ---------- Chat Panel ----------
   225→function ChatPanel({ user }: { user: User }) {
   226→  const [messages, setMessages] = useState<ChatMessage[]>([
   227→    {
   228→      role: 'assistant',
   229→      content: `Hi ${user.name.split(' ')[0]}! I'm Labby, your lab booking assistant. I can help you book a computer lab, check availability, view your bookings, or cancel one. What would you like to do?`,
   230→      ts: Date.now(),
   231→    },
   232→  ])
   233→  const [input, setInput] = useState('')
   234→  const [loading, setLoading] = useState(false)
   235→  const scrollRef = useRef<HTMLDivElement>(null)
   236→  const { toast } = useToast()
   237→
   238→  useEffect(() => {
   239→    scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight, behavior: 'smooth' })
   240→  }, [messages, loading])
   241→
   242→  const send = async (text: string) => {
   243→    const trimmed = text.trim()
   244→    if (!trimmed || loading) return
   245→    const history = messages.map((m) => ({ role: m.role, content: m.content }))
   246→    const next: ChatMessage[] = [...messages, { role: 'user', content: trimmed, ts: Date.now() }]
   247→    setMessages(next)
   248→    setInput('')
   249→    setLoading(true)
   250→    try {
   251→      const res = await fetch('/api/chat', {
   252→        method: 'POST',
   253→        headers: { 'Content-Type': 'application/json' },
   254→        body: JSON.stringify({ userId: user.id, message: trimmed, history }),
   255→      })
   256→      const data = await res.json()
   257→      if (!res.ok) throw new Error(data.error || 'Chat failed')
   258→      setMessages((prev) => [...prev, { role: 'assistant', content: data.reply, ts: Date.now() }])
   259→    } catch (e: any) {
   260→      toast({ title: 'Chat error', description: e.message, variant: 'destructive' })
   261→      setMessages((prev) => [
   262→        ...prev,
   263→        { role: 'assistant', content: `Sorry, I hit an error: ${e.message}. Please try again.`, ts: Date.now() },
   264→      ])
   265→    } finally {
   266→      setLoading(false)
   267→    }
   268→  }
   269→
   270→  return (
   271→    <div className="flex flex-col h-full">
   272→      <div className="flex-1 overflow-hidden">
   273→        <ScrollArea className="h-full" ref={scrollRef as any}>
   274→          <div className="space-y-4 p-4 max-w-3xl mx-auto">
   275→            {messages.map((m, i) => (
   276→              <MessageBubble key={i} message={m} userName={user.name} />
   277→            ))}
   278→            {loading && (
   279→              <div className="flex gap-3 items-start">
   280→                <div className="w-8 h-8 rounded-full bg-gradient-to-br from-emerald-500 to-teal-600 text-white flex items-center justify-center shrink-0">
   281→                  <Bot className="w-4 h-4" />
   282→                </div>
   283→                <div className="bg-muted rounded-2xl rounded-tl-sm px-4 py-3 flex items-center gap-2">
   284→                  <Loader2 className="w-4 h-4 animate-spin" />
   285→                  <span className="text-sm text-muted-foreground">Thinking…</span>
   286→                </div>
   287→              </div>
   288→            )}
   289→          </div>
   290→        </ScrollArea>
   291→      </div>
   292→
   293→      <div className="border-t bg-background/80 backdrop-blur">
   294→        <div className="max-w-3xl mx-auto p-4 space-y-3">
   295→          <div className="flex flex-wrap gap-2">
   296→            {QUICK_PROMPTS.map((p) => (
   297→              <Button
   298→                key={p}
   299→                variant="outline"
   300→                size="sm"
   301→                className="text-xs h-8"
   302→                disabled={loading}
   303→                onClick={() => send(p)}
   304→              >
   305→                {p}
   306→              </Button>
   307→            ))}
   308→          </div>
   309→          <form
   310→            onSubmit={(e) => {
   311→              e.preventDefault()
   312→              send(input)
   313→            }}
   314→            className="flex gap-2"
   315→          >
   316→            <Input
   317→              value={input}
   318→              onChange={(e) => setInput(e.target.value)}
   319→              placeholder="Ask Labby anything — e.g. 'book Lab A tomorrow 2-4pm'"
   320→              disabled={loading}
   321→              className="flex-1"
   322→            />
   323→            <Button type="submit" disabled={loading || !input.trim()} size="icon" className="bg-emerald-600 hover:bg-emerald-700 text-white">
   324→              {loading ? <Loader2 className="w-4 h-4 animate-spin" /> : <Send className="w-4 h-4" />}
   325→            </Button>
   326→          </form>
   327→        </div>
   328→      </div>
   329→    </div>
   330→  )
   331→}

   333→function MessageBubble({ message, userName }: { message: ChatMessage; userName: string }) {
   334→  const isUser = message.role === 'user'
   335→  const initials = isUser ? userName.split(' ').map((s) => s[0]).slice(0, 2).join('') : 'L'
   336→
   337→  return (
   338→    <div className={`flex gap-3 items-start ${isUser ? 'flex-row-reverse' : ''}`}>
   339→      <div
   340→        className={`w-8 h-8 rounded-full flex items-center justify-center shrink-0 text-white text-xs font-semibold ${
   341→          isUser ? 'bg-slate-700' : 'bg-gradient-to-br from-emerald-500 to-teal-600'
   342→        }`}
   343→      >
   344→        {isUser ? initials : <Bot className="w-4 h-4" />}
   345→      </div>
   346→      <div
   347→        className={`max-w-[80%] px-4 py-3 rounded-2xl text-sm whitespace-pre-wrap ${
   348→          isUser ? 'bg-slate-700 text-white rounded-tr-sm' : 'bg-muted rounded-tl-sm'
   349→        }`}
   350→      >
   351→        <MarkdownLite content={message.content} />
   352→      </div>
   353→    </div>
   354→  )
   355→}

   357→// Lightweight markdown renderer — handles **bold**, `code`, and bullet lists
   358→function MarkdownLite({ content }: { content: string }) {
   359→  const lines = content.split('\n')
   360→  return (
   361→    <div className="space-y-1">
   362→      {lines.map((line, i) => {
   363→        if (line.trim().startsWith('• ') || line.trim().startsWith('- ')) {
   364→          return (
   365→            <div key={i} className="flex gap-2 pl-1">
   366→              <span className="text-emerald-500 font-bold">•</span>
   367→              <span>{renderInline(line.trim().slice(2))}</span>
   368→            </div>
   369→          )
   370→        }
   371→        return <div key={i}>{renderInline(line) || <br />}</div>
   372→      })}
   373→    </div>
   374→  )
   375→}
   376→
   377→function renderInline(text: string) {
   378→  if (!text) return null
   379→  const parts = text.split(/(\*\*[^*]+\*\*|`[^`]+`)/g)
   380→  return parts.map((p, i) => {
   381→    if (p.startsWith('**') && p.endsWith('**')) {
   382→      return <strong key={i} className="font-semibold">{p.slice(2, -2)}</strong>
   383→    }
   384→    if (p.startsWith('`') && p.endsWith('`')) {
   385→      return <code key={i} className="px-1.5 py-0.5 rounded bg-background/60 text-xs font-mono">{p.slice(1, -1)}</code>
   386→    }
   387→    return <span key={i}>{p}</span>
   388→  })
   389→}

   515→// ---------- My Bookings Panel ----------
   516→function MyBookingsPanel({ user }: { user: User }) {
   517→  const [bookings, setBookings] = useState<Booking[]>([])
   518→  const [loading, setLoading] = useState(true)
   519→  const { toast } = useToast()
   520→
   521→  const load = useCallback(async () => {
   522→    setLoading(true)
   523→    try {
   524→      const res = await fetch(`/api/bookings?userId=${user.id}&scope=mine`)
   525→      const data = await res.json()
   526→      if (!res.ok) throw new Error(data.error)
   527→      setBookings(data.bookings || [])
   528→    } catch (e: any) {
   529→      toast({ title: 'Failed to load bookings', description: e.message, variant: 'destructive' })
   530→    } finally {
   531→      setLoading(false)
   532→    }
   533→  }, [user.id, toast])
   534→
   535→  useEffect(() => {
   536→    load()
   537→  }, [load])
   538→
   539→  const cancel = async (id: string) => {
   540→    try {
   541→      const res = await fetch(`/api/bookings/${id}?userId=${user.id}`, { method: 'DELETE' })
   542→      const data = await res.json()
   543→      if (!res.ok) throw new Error(data.error)
   544→      toast({ title: 'Booking cancelled', description: 'The slot is now free for others.' })
   545→      load()
   546→    } catch (e: any) {
   547→      toast({ title: 'Failed to cancel', description: e.message, variant: 'destructive' })
   548→    }
   549→  }
   550→
   551→  const today = new Date().toISOString().slice(0, 10)
   552→  const upcoming = bookings.filter((b) => b.date >= today)
   553→  const past = bookings.filter((b) => b.date < today)
   554→
   555→  return (
   556→    <div className="space-y-4 p-4 max-w-5xl mx-auto">
   557→      <Card>
   558→        <CardHeader>
   559→          <CardTitle className="flex items-center gap-2"><LayoutDashboard className="w-5 h-5 text-emerald-600" /> My bookings</CardTitle>
   560→          <CardDescription>Your upcoming and past lab reservations.</CardDescription>
   561→        </CardHeader>
   562→        <CardContent>
   563→          {loading ? (
   564→            <div className="flex items-center justify-center py-8"><Loader2 className="w-6 h-6 animate-spin text-muted-foreground" /></div>
   565→          ) : bookings.length === 0 ? (
   566→            <div className="text-center py-12 text-muted-foreground">
   567→              <CalendarIcon className="w-10 h-10 mx-auto mb-3 opacity-50" />
   568→              <p>You don't have any bookings yet.</p>
   569→              <p className="text-sm mt-1">Use the Chat tab to ask Labby to book a lab for you.</p>
   570→            </div>
   571→          ) : (
   572→            <div className="space-y-6">
   573→              <div>
   574→                <h3 className="text-sm font-semibold mb-2 text-muted-foreground uppercase tracking-wide">Upcoming ({upcoming.length})</h3>
   575→                {upcoming.length === 0 ? (
   576→                  <p className="text-sm text-muted-foreground">No upcoming bookings.</p>
   577→                ) : (
   578→                  <div className="space-y-2">
   579→                    {upcoming.map((b) => (
   580→                      <BookingCard key={b.id} booking={b} onCancel={() => cancel(b.id)} />
   581→                    ))}
   582→                  </div>
   583→                )}
   584→              </div>
   585→              {past.length > 0 && (
   586→                <div>
   587→                  <h3 className="text-sm font-semibold mb-2 text-muted-foreground uppercase tracking-wide">Past ({past.length})</h3>
   588→                  <div className="space-y-2 opacity-70">
   589→                    {past.map((b) => (
   590→                      <BookingCard key={b.id} booking={b} onCancel={() => cancel(b.id)} />
   591→                    ))}
   592→                  </div>
   593→                </div>
   594→              )}
   595→            </div>
   596→          )}
   597→        </CardContent>
   598→      </Card>
   599→    </div>
   600→  )
